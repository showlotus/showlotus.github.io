```js
/**
 * 获取可用端口号，如果当前端口号已被占用，则自增，直到找到可用端口号为止
 */
const getAvailablePort = port => {
  let targetPort = port
  while (true && targetPort < 9000) {
    try {
      execSync(`netstat -aon | findStr "${targetPort}"`)
      targetPort++
    } catch (e) {
      return targetPort
    }
  }
}
```

用法：

```js
// xx.config.js
module.exports = {
  devServer: {
    port: getAvailablePort(8080)
  }
}
```
