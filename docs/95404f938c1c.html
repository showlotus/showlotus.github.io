const Color = less.color
const { fade, shade, tint } = less.functions.functionRegistry._data

function toRgb(color) {
  const { alpha, rgb } = color
  const [r, g, b] = rgb.map(v => parseInt(v, 10))
  return `rgba(${r}, ${g}, ${b}, ${alpha})`
}
